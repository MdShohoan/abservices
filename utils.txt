    # Check if the object has these attributes

            closing = accstatusresult.closing if hasattr(accstatusresult, 'closing') else False

            print(f"blocked : {blocked}")
            print(f"closing : {closing}")
            print(f"inactive: {inactive}")
            print(f"decOrLiq: {decOrLiq}")
            print('==================================')
            
            if inactive == True:
                return True  # Account is inactive
            else:
                return False  # Account is active
        else:
            return True  # Default to True if we can't determine
    
    except Exception as e:
        print(f"Error accessing account status attributes: {e}")
        return True  # Default to True on error
def customerDetails(custno):
    cusresult = {}
    
    try:
        sessiontoken = getEQSessionToken()
        if not sessiontoken:
            LOG.error("Failed to get session token for customerDetails")
            return {'success': False, 'retMessage': 'Session token failed'}
        
        request_data = { 
            'sessionToken': sessiontoken,
            'custno': custno
        }
        
        LOG.info('Customer: %s: Request Data: session token: %s', custno, sessiontoken)
        transactionClient = Client(eqEnqurywsdl)
        
        print('======= CALLING customerDetails API =======')
        print(f"custno being sent: '{custno}'")
        
        # Make the API call
        result = transactionClient.service.customerDetails(**request_data)
        
        print('======= customerDetails RESULT =======')
        print(result)
        print('======================================')
        
        # Convert Zeep object to dictionary if needed
        if hasattr(result, '__dict__'):
            # It's a Zeep object - convert to dictionary
            cusresult = {}
            for key in dir(result):
                if not key.startswith('_') and not callable(getattr(result, key)):
                    value = getattr(result, key)
                    # Handle None values
                    if value is None:
                        cusresult[key] = ""
                    else:
                        cusresult[key] = value
            
            # Also ensure success field is properly set
            if 'success' not in cusresult and hasattr(result, 'success'):
                cusresult['success'] = result.success
        elif isinstance(result, dict):
            # It's already a dictionary
            cusresult = result
        else:
            # Unknown type
            LOG.error("Unexpected result type from customerDetails: %s", type(result))
            cusresult = {'success': False, 'retMessage': 'Unexpected response type'}
        
        # Ensure all required fields exist with defaults
        default_fields = {
            'success': False,
            'retMessage': '',
            'customerName': '',
            'mobile': '',
            'email1': '',
            'email2': '',
            'basicNumber': '',
            'businessPhone': '',
            'customerType': '',
            'customerTypeDesc': '',
            'dateOfBirth': '',
            'defaultBranch': '',
            'homePhone': '',
            'idDocCode': '',
            'idDocNo': '',
            'idExpiryDate': '',
            'idIssueDate': '',
            'idIssuer': '',
            'residenceCountry': '',
            'taxIdNo': ''
        }
        
        # Merge with defaults to ensure all fields exist
        for field, default_value in default_fields.items():
            if field not in cusresult or cusresult[field] is None:
                cusresult[field] = default_value
        
        LOG.info('Customer: %s: Customer Details Result: %s', custno, cusresult)
        
    except Exception as error:
        print(f"Error in customerDetails: {error}")
        LOG.error("***********Error in customerDetails: %s for custno: %s***********", str(error), custno)
        cusresult = {
            'success': False,
            'retMessage': str(error),
            'error': str(error)
        }
    
    return cusresult


def customerDetailsWithRetry(custno, branch_code=None):
    """
    Try multiple customer number formats to find the correct one
    """
    # Define possible customer number formats to try
    possible_custnos = [custno]
    
    # Add formats with branch code if provided
    if branch_code:
        possible_custnos.append(f"{branch_code}{custno}")
        possible_custnos.append(f"{custno}{branch_code}")
    
    # Add padded versions
    possible_custnos.append(custno.zfill(10))
    possible_custnos.append(custno.zfill(8))
    
    # Remove duplicates while preserving order
    seen = set()
    possible_custnos = [x for x in possible_custnos if not (x in seen or seen.add(x))]
    
    for test_custno in possible_custnos:
        print(f"Trying customerDetails with custno: '{test_custno}'")
        result = customerDetails(test_custno)
        
        # Check if successful
        if isinstance(result, dict) and result.get('success') == True:
            print(f"Success with custno: '{test_custno}'")
            return result
        
        # Also check if we got valid data even if success flag is not set
        if isinstance(result, dict) and result.get('customerName'):
            print(f"Got customer data with custno: '{test_custno}'")
            result['success'] = True
            return result
    
    print("All customer number formats failed")
    return {'success': False, 'retMessage': 'No valid customer number format found'}
